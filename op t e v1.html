Quick Start 快速上手 


# pip install openai
from openai import OpenAI

client = OpenAI(api_key="YOUR_API_KEY")
resp = client.chat.completions.create(
  model="o4-mini",
  messages=[{"role":"user","content":"Explain quantum computing in simple terms"}]
)
print(resp.choices[0].message.content)
